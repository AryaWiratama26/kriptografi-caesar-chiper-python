Fungsi CAESAR_ENCRYPT(kata, keys)
    HASIL_ENCRYPT = ""
    
    Untuk setiap KARAKTER dalam kata LAKUKAN:
        Jika KARAKTER adalah huruf kapital ('A' sampai 'Z') MAKA:
            // Geser untuk huruf kapital: (Kode_ASCII - Kode_A + kunci) modulus 26 + Kode_A
            INDEKS_BARU = (Kode_ASCII(KARAKTER) - Kode_ASCII('A') + keys) MOD 26
            KARAKTER_TERENKRIPSI = Karakter_dari_Kode_ASCII(INDEKS_BARU + Kode_ASCII('A'))
            HASIL_ENCRYPT = HASIL_ENCRYPT + KARAKTER_TERENKRIPSI
            
        Jika KARAKTER adalah huruf kecil ('a' sampai 'z') MAKA:
            // Geser untuk huruf kecil: (Kode_ASCII - Kode_a + kunci) modulus 26 + Kode_a
            INDEKS_BARU = (Kode_ASCII(KARAKTER) - Kode_ASCII('a') + keys) MOD 26
            KARAKTER_TERENKRIPSI = Karakter_dari_Kode_ASCII(INDEKS_BARU + Kode_ASCII('a'))
            HASIL_ENCRYPT = HASIL_ENCRYPT + KARAKTER_TERENKRIPSI
            
        Jika TIDAK (karakter adalah simbol, angka, spasi, dll.) MAKA:
            HASIL_ENCRYPT = HASIL_ENCRYPT + KARAKTER
            
    Kembalikan HASIL_ENCRYPT
Akhir Fungsi

Fungsi CAESAR_DECRYPT(kata, keys)
    // Dekripsi = Enkripsi dengan kunci (26 - keys)
    Kembalikan CAESAR_ENCRYPT(kata, 26 - keys)
Akhir Fungsi
